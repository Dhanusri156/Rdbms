    CREATE OR REPLACE TRIGGER check_room_availability  
BEFORE INSERT OR UPDATE ON Booking  
FOR EACH ROW 
 DECLARE 
room_count integer;  
BEGIN  
SELECT COUNT(*) INTO room_count  
 FROM Rooms 
WHERE room_id = :NEW.room_id  
AND Status = 'Vacant';  
if room_count = 0 THEN  
RAISE_APPLICATION_ERROR(-20001,'Selected room is not available for the given date range.');  
END if;  
if room_count > 0 THEN  
DBMS_OUTPUT.PUT_LINE('Rooms are available for the given date range.');  
ELSE 
DBMS_OUTPUT.PUT_LINE('No rooms available for the given date range.');  
END if;  
END;
